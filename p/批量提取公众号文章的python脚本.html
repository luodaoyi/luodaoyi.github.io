import json
import pprint

links_array = []
with open('msg.txt',encoding='utf-8') as file:
	for line in file:
		s = json.loads(line)
		[links_array.append(i['link']) for i in s['app_msg_list']]
		
with open('link.txt','w') as f :
	[f.write(l + '\n') for l in links_array]
